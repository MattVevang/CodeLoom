// Copyright (c) FIRST and other WPILib contributors.
// Open Source Software; you can modify and/or share it under the terms of
// the WPILib BSD license file in the root directory of this project.

package frc.robot.systems.drive;

import com.chargerrobotics.common.subsystems.drivetrains.swerve.SwerveSubsystem;
import com.chargerrobotics.common.subsystems.drivetrains.swerve.SwerveSubsystem.DriveState;
import edu.wpi.first.math.controller.PIDController;
import edu.wpi.first.math.geometry.Pose2d;
import edu.wpi.first.math.kinematics.ChassisSpeeds;
import edu.wpi.first.wpilibj2.command.SubsystemBase;
import java.io.IOException;
import java.io.UncheckedIOException;
import java.util.function.DoubleSupplier;
import lombok.Builder;
import lombok.Getter;

public class DriveSystem extends SubsystemBase {
  @Getter private SwerveSubsystem swerveSubsystem;

  public static final double MAX_SPEED_METERS_PER_SECOND = 2.0;
  public static final double TRANSLATION_AUTO_P = 0.004;
  public static final double TRANSLATION_AUTO_I = 0;
  public static final double TRANSLATION_AUTO_D = 0;
  public static final double TRANSLATION_PID_TOLERANCE = 0.01;
  public static final double TRANSLATION_PID_TOLERANCE_METERS_PER_SECOND = 0.02;
  public static final double ROTATIONAL_PID_TOLERANCE = 0.01;

  private DoubleSupplier forwardTranslationVelocity;
  private DoubleSupplier leftTranslationVelocity;
  private DoubleSupplier leftHeadingVelocity;
  private PIDController forwardPidController;
  private PIDController leftPidController;

  /** Creates a new DriveSystem. */
  @Builder
  public DriveSystem(
      DoubleSupplier translationForward,
      DoubleSupplier translationLeft,
      DoubleSupplier rotationScalar) {
    try {
      swerveSubsystem = SwerveSubsystem.getInstance(MAX_SPEED_METERS_PER_SECOND);
    } catch (IOException e) {
      throw new UncheckedIOException(e);
    }

    this.forwardTranslationVelocity = translationForward;
    this.leftTranslationVelocity = translationLeft;
    this.leftHeadingVelocity = rotationScalar;

    this.forwardPidController =
        new PIDController(TRANSLATION_AUTO_P, TRANSLATION_AUTO_I, TRANSLATION_AUTO_D);
    this.leftPidController =
        new PIDController(TRANSLATION_AUTO_P, TRANSLATION_AUTO_I, TRANSLATION_AUTO_D);
    this.forwardPidController.setTolerance(
        TRANSLATION_PID_TOLERANCE, TRANSLATION_PID_TOLERANCE_METERS_PER_SECOND);
    this.leftPidController.setTolerance(
        TRANSLATION_PID_TOLERANCE, TRANSLATION_PID_TOLERANCE_METERS_PER_SECOND);
    swerveSubsystem
        .getChassisStates()
        .getSwerveController()
        .thetaController
        .setTolerance(ROTATIONAL_PID_TOLERANCE);
  }

  public void moveToPose(Pose2d pose) {
    swerveSubsystem.setChassisSpeedsFunction(
        states ->
            new ChassisSpeeds(
                forwardPidController.calculate(states.getFieldPose().getX(), pose.getX()),
                leftPidController.calculate(states.getFieldPose().getY(), pose.getY()),
                swerveSubsystem
                    .getChassisStates()
                    .getSwerveController()
                    .thetaController
                    .calculate(
                        states.getFieldPose().getRotation().getRadians(),
                        pose.getRotation().getRadians())));
    swerveSubsystem.setDriveState(DriveState.FIELD_ORIENTED);
  }

  public void setFieldOriented() {
    swerveSubsystem.setChassisSpeedsFunction(
        states ->
            states
                .getSwerveController()
                .getRawTargetSpeeds(
                    forwardTranslationVelocity.getAsDouble() * states.getMaxTranslationalSpeed(),
                    leftTranslationVelocity.getAsDouble() * states.getMaxTranslationalSpeed(),
                    leftHeadingVelocity.getAsDouble() * states.getMaxRotationalSpeed()));
    swerveSubsystem.setDriveState(DriveState.FIELD_ORIENTED);
  }

  public Pose2d getFieldPose() {
    return swerveSubsystem.getChassisStates().getFieldPose();
  }

  public ChassisSpeeds getRobotVelocity() {
    return swerveSubsystem.getChassisStates().getChassisSpeeds();
  }

  public ChassisSpeeds getFieldVelocity() {
    return ChassisSpeeds.fromRobotRelativeSpeeds(getRobotVelocity(), getFieldPose().getRotation());
  }
}

   * @param gearing The reduction between motor and drum, as a ratio of output to input.
   * @param p The p value of pid controller.
   * @param i The i value of pid controller.
   * @param d The d value of pid controller.
   * @param dtSeconds The time between updates for the simulation.
   */
  public BeltedIntakeIoSim(
      int numMotors,
      double jKgMetersSquared,
      double gearing,
      double p,
      double i,
      double d,
      double dtSeconds) {
    this.dtSeconds = dtSeconds;
    velocityPidController = new PIDController(p, i, d);
    positionPidController = new PIDController(p, i, d);
    sim =
        new DCMotorSim(
            LinearSystemId.createDCMotorSystem(
                DCMotor.getNEO(numMotors), jKgMetersSquared, gearing),
            DCMotor.getNEO(numMotors));
  }

  @Override
  public void updateInputs(BeltedIntakeIoInputs inputs) {
    if (isPositionControl) {
      double voltage =
          positionPidController.calculate(sim.getAngularPositionRad(), newPositionRadians);
      sim.setInputVoltage(voltage);
    } else {
      double voltage = velocityPidController.calculate(sim.getAngularVelocityRPM(), newVelocityRpm);
      sim.setInputVoltage(voltage);
    }
    sim.update(dtSeconds);

    inputs.setMotorPositionRadians(sim.getAngularPositionRad());
    inputs.setMotorVelocityRpm(sim.getAngularVelocityRPM());
    inputs.setMotorAppliedVoltage(sim.getInputVoltage());
    inputs.setMotorCurrentAmps(sim.getCurrentDrawAmps());
  }

  @Override
  public void setVelocityTarget(double velocityRpm) {
    isPositionControl = false;
    newVelocityRpm = velocityRpm;
  }

  @Override
  public void setPositionTarget(double positionRadians) {
    isPositionControl = true;
    newPositionRadians = positionRadians;
  }

  @Override
  public void resetMotorPositionTo(double newPositionRadians) {
    isPositionControl = true;
    this.newPositionRadians = newPositionRadians;
  }

  @Override
  public void setMotorToIdle() {
    newVelocityRpm = 0.0;
  }
}

```text
package frc.robot.systems.intake;

import com.chargerrobotics.common.subsystem.belted_intake.BeltedIntakeSubsystem;
import com.chargerrobotics.common.subsystem.belted_intake.BeltedIntakeSubsystem.BeltedIntakeSubsystemState;
import com.chargerrobotics.common.subsystems.singledofarm.SingleDofArmSubsystem;
import com.chargerrobotics.common.subsystems.singledofarm.SingleDofArmSubsystem.SingleDofArmStates;
import edu.wpi.first.wpilibj2.command.SubsystemBase;
import lombok.Builder;
import lombok.Getter;
import lombok.NonNull;
import lombok.Setter;
import org.littletonrobotics.junction.Logger;

/** The IntakeSystem can extend and retract the collector to intake or eject fuel. */
public class IntakeSystem extends SubsystemBase {
  private SingleDofArmSubsystem<IntakeInternalStates> armSubsystem;
  private BeltedIntakeSubsystem hopperSubsystem;
  private BeltedIntakeSubsystem intakeSubsystem;
  @Getter private IntakeStates currentIntakeState;
  @Setter private IntakeStates desiredIntakeState;

  /**
   * Constructor of IntakeSystem.
   *
   * @param intakeSubsystem A subsystem that handles the control of the fuel.
   * @param hopperSubsystem A subsystem that handles the creation and control of the Hopper.
   */
  @Builder
  public IntakeSystem(
      @NonNull BeltedIntakeSubsystem intakeSubsystem,
      @NonNull BeltedIntakeSubsystem hopperSubsystem,
      @NonNull SingleDofArmSubsystem<IntakeInternalStates> armSubsystem) {
    this.intakeSubsystem = intakeSubsystem;
    this.hopperSubsystem = hopperSubsystem;
    this.armSubsystem = armSubsystem;
    setDesiredIntakeState(IntakeStates.IDLE);
  }

  public enum IntakeStates {
    /** State for currently doing nothing. */
    IDLE,
    /** State for when fuel needs to be taken out. */
    EJECT,
    /** State for when the intake is angled to take in fuel. */
    INTAKE,
    /** State for when it is retracted and still intaking. */
    AGITATION
  }

  /** The internal states of {@linkplain IntakeStates}. */
  public enum IntakeInternalStates {
    /** State for when the intake is retracted. */
    RETRACTED,
    /** State for when the intake is extended. */
    EXTENDED
  }

  // Runs 50 times every second (roughly) and drives the state machine.
  @Override
  public void periodic() {
    BeltedIntakeSubsystemState currentIntakeSubsystemState = intakeSubsystem.getCurrentState();
    BeltedIntakeSubsystemState currentHopperState = hopperSubsystem.getCurrentState();
    SingleDofArmStates currentArmState = armSubsystem.getCurrentState();

    switch (desiredIntakeState) {
      case IDLE:
        intakeSubsystem.setDesiredState(BeltedIntakeSubsystemState.STOPPED);
        hopperSubsystem.setDesiredState(BeltedIntakeSubsystemState.STOPPED);
        armSubsystem.setPositionControl(IntakeInternalStates.RETRACTED);
        if (currentIntakeSubsystemState == BeltedIntakeSubsystemState.STOPPED
            && currentHopperState == BeltedIntakeSubsystemState.STOPPED
            && currentArmState == SingleDofArmStates.AT_SETPOINT) {
          armSubsystem.setMotorToIdle();
          currentIntakeState = IntakeStates.IDLE;
        }
        break;
      case EJECT:
        intakeSubsystem.setDesiredState(BeltedIntakeSubsystemState.EJECT);
        hopperSubsystem.setDesiredState(BeltedIntakeSubsystemState.EJECT);
        armSubsystem.setPositionControl(IntakeInternalStates.EXTENDED);
        if (currentIntakeSubsystemState == BeltedIntakeSubsystemState.EJECT
            && currentHopperState == BeltedIntakeSubsystemState.EJECT
            && currentArmState == SingleDofArmStates.AT_SETPOINT) {
          currentIntakeState = IntakeStates.EJECT;
        }
        break;
      case INTAKE:
        intakeSubsystem.setDesiredState(BeltedIntakeSubsystemState.INTAKE);
        hopperSubsystem.setDesiredState(BeltedIntakeSubsystemState.INTAKE);
        armSubsystem.setPositionControl(IntakeInternalStates.EXTENDED);
        if (currentIntakeSubsystemState == BeltedIntakeSubsystemState.INTAKE
            && currentHopperState == BeltedIntakeSubsystemState.INTAKE
            && currentArmState == SingleDofArmStates.AT_SETPOINT) {
          currentIntakeState = IntakeStates.INTAKE;
        }
        break;
      case AGITATION:
        intakeSubsystem.setDesiredState(BeltedIntakeSubsystemState.INTAKE);
        hopperSubsystem.setDesiredState(BeltedIntakeSubsystemState.INTAKE);
        armSubsystem.setPositionControl(IntakeInternalStates.RETRACTED);
        if (currentIntakeSubsystemState == BeltedIntakeSubsystemState.INTAKE
            && currentHopperState == BeltedIntakeSubsystemState.INTAKE
            && currentArmState == SingleDofArmStates.AT_SETPOINT) {
          currentIntakeState = IntakeStates.AGITATION;
        }
        break;
    }
    Logger.recordOutput("IntakeSystem/CurrentState", currentIntakeState);
    Logger.recordOutput("IntakeSystem/DesiredState", desiredIntakeState);
  }
}

/** Create a mock flywheel so that we can simulate the robot with advantageKit */
public class FlywheelIoSimulation implements FlywheelIo {

  private DCMotorSim sim;
  private PIDController pid;
  private double targetRpm = 0.0;
  double feedForwardVolts = 0.0;
  private double dtSeconds;
  private SimpleMotorFeedforward feedForward = new SimpleMotorFeedforward(0, 1);

  public FlywheelIoSimulation(FlywheelSimIoConfiguration configuration) {
    this.dtSeconds = configuration.getDtSecond();
    LinearSystem<N2, N1, N2> plant =
        LinearSystemId.createDCMotorSystem(
            DCMotor.getNEO(configuration.getNumMotors()),
            configuration.getJkgMetersSquared(),
            configuration.getGearing());
    sim =
        new DCMotorSim(
            plant,
            DCMotor.getNEO(configuration.getNumMotors()),
            configuration.getMeasurementStdDevs(),
            configuration.getMeasurementStdDevs());

  }

  @Override
  public void updateInputs(FlywheelIoInput inputs) {
    sim.update(dtSeconds);
    double currentRPM = sim.getAngularVelocityRPM();
    double pidOutput = pid.calculate(currentRPM, targetRpm) + feedForwardVolts;
    sim.setInputVoltage(pidOutput);

    inputs.motorVelocityRpm = targetRpm;
    inputs.motorAppliedVoltage = sim.getInputVoltage();
    inputs.motorCurrentAmps = sim.getCurrentDrawAmps();
  }

  @Override
  public void setVelocityTarget(double velocityRpm) {
    targetRpm = velocityRpm;
    feedForwardVolts = 0.0;
  }

  @Override
  public void setVoltage(double volts) {
    sim.setInputVoltage(volts);
    feedForwardVolts = 0.0;
  }

  @Override
  public double getVelocity() {
    return sim.getAngularVelocityRPM();
  }

  @Override
  public void setMaxVelocity(double velocityRpm) {
    targetRpm = velocityRpm;
    feedForwardVolts = feedForward.calculate(velocityRpm);
  }

  @Override
  public void setMotorToIdle() {
    setVoltage(0);
  }
}

package frc.robot.systems.shooter.flywheel;

import edu.wpi.first.units.Units;
import edu.wpi.first.units.measure.AngularVelocity;
import edu.wpi.first.wpilibj2.command.SubsystemBase;
import lombok.Getter;
import org.littletonrobotics.junction.Logger;

/**
 * Handles all control of the robot's flywheel including spinning it up, checking its speed, and
 * managing its state for shooting.
 */
public class FlywheelSubsystem extends SubsystemBase {

  public enum FlywheelSubsystemState {
    /** When motor stopped. */
    IDLE,
    /** When the speed is increase but not ready yet. */
    ACTIVATING,
    /** When reached the target speed. */
    READY
  }

  public FlywheelSubsystem(FlywheelIo io, String subsystemName) {
    this.io = io;
    this.subsystemName = subsystemName;
    currentState = FlywheelSubsystemState.IDLE;
  }

  private final FlywheelIo io;
  private final FlywheelIoInputAutoLogged inputs = new FlywheelIoInputAutoLogged();

  @Getter private FlywheelSubsystemState currentState;

  private double targetSpeed = 0.0;
  private double speedTolerance = 50.0;
  private String subsystemName;

  public void setMotorToIdle() {
    io.setMotorToIdle();
  }

  public void setMotorSpeed(AngularVelocity velocity) {
    targetSpeed = velocity.in(Units.RPM);
    io.setVelocityTarget(targetSpeed);
  }

  @Override
  public void periodic() {
    io.updateInputs(inputs);
    Logger.processInputs(subsystemName, inputs);
    findCurrentState();
  }

  private void findCurrentState() {
    double currentVelocity = inputs.getMotorVelocityRpm();
    double currentVoltage = inputs.getMotorAppliedVoltage();

    if (currentVoltage == 0.0) {
      currentState = FlywheelSubsystemState.IDLE;
    } else if (Math.abs(targetSpeed - currentVelocity) < speedTolerance) {
      currentState = FlywheelSubsystemState.READY;
    } else {
      currentState = FlywheelSubsystemState.ACTIVATING;
    }
  }

  public double exitVelocity() {
    return (4.0 * inputs.getMotorVelocityRpm()) / 2;
  }
}

    /** State for doing nothing */
    IDLE,
    /** Target is reached state */
    AT_SETPOINT,
    /** Transitional state */
    MOVING_TO_SETPOINT,
    /** State for calibrating the hood */
    CALIBRATE
  }

  public enum HoodCalibrateStates {
    NOT_CALIBRATED,
    CALIBRATING,
    CALIBRATED
  }

  @Getter private HoodSubsystemStates currentState;
  @Getter private HoodCalibrateStates currentCalibrateState;
  @Getter private boolean isCalibrated;
  @Setter @Getter private Angle desiredAngle;
  private HoodIo io;
  private HoodIoInputAutoLogged inputs;

  /** Creates a new HoodSubsystem. */
  public HoodSubsystem(HoodIo io) {
    currentState = HoodSubsystemStates.IDLE;
    this.io = io;
    inputs = new HoodIoInputAutoLogged();
    calibrate();
  }

  public void calibrate() {
    isCalibrated = false;
    currentCalibrateState = HoodCalibrateStates.NOT_CALIBRATED;
  }

  public void setAngle(Angle newAngle) {
    desiredAngle = newAngle;
    io.setPositionControl(newAngle);
  }

  public void stopMotor() {
    io.stopMotor();
  }

  /** Gets called 50 times per second + drives state machine */
  @Override
  public void periodic() {
    io.updateInputs(inputs);
    Logger.processInputs("HoodSubsystem", inputs);
    if (desiredAngle == null) {
      // This is for first time running the subsystem as the start desiredAngle is null.
      stopMotor();
      currentState = HoodSubsystemStates.IDLE;
    } else if (inputs.getMotorVoltage() == Units.Volts.of(0.0)) {
      currentState = HoodSubsystemStates.IDLE;
    } else if (desiredAngle.isNear(inputs.getMotorPosition(), Units.Rotations.of(1.0))) {
      currentState = HoodSubsystemStates.AT_SETPOINT;
    } else {
      currentState = HoodSubsystemStates.MOVING_TO_SETPOINT;
    }
    Logger.recordOutput("HoodSubsystem/CurrentState", currentState);
    Logger.recordOutput("HoodSubsystem/DesiredAngle", desiredAngle);
  }

public class TurretSimIo implements TurretIo {
  private DCMotorSim motorSim;
  private PIDController positionController;
  private Time dt;
  private Angle newPosition;
  private DCMotor gearboxNEO;

  /**
   * Constructor for TurretSimIo.
   *
   * <p>Creates a simulated DC motor mechanism.
   *
   * @param numMotors The number of motors in the gearbox.
   * @param j The moment of inertia J of the DC motor.
   * @param gearing The reduction between motor and drum, as a ratio of output to input.
   * @param p The p value of pid controller.
   * @param i The i value of pid controller.
   * @param d The d value of pid controller.
   * @param dt The time between updates for the simulation.
   */
  public TurretSimIo(
      int numMotors, MomentOfInertia j, double gearing, double p, double i, double d, Time dt) {
    newPosition = Units.Rotations.of(0.0);
    this.dt = dt;
    gearboxNEO = DCMotor.getNEO(numMotors);
    positionController = new PIDController(p, i, d);
    motorSim =
        new DCMotorSim(
            LinearSystemId.createDCMotorSystem(
                gearboxNEO, j.in(Units.KilogramSquareMeters), gearing),
            gearboxNEO);
  }

  @Override
  public void updateInputs(TurretSubsystemInputs inputs) {
    double voltage =
        positionController.calculate(
            motorSim.getAngularPositionRotations(), newPosition.in(Units.Rotations));
    motorSim.setInputVoltage(voltage);
    motorSim.update(dt.in(Units.Seconds));

    inputs.setMotorCurrent(Units.Amps.of(motorSim.getCurrentDrawAmps()));
    inputs.setMotorPosition(motorSim.getAngularPosition());
    inputs.setMotorPositionSetpoint(newPosition);
    inputs.setMotorVelocity(motorSim.getAngularVelocity());
    inputs.setMotorVoltage(Units.Volts.of(motorSim.getInputVoltage()));
  }

  @Override
  public void setPositionControl(Angle value) {
    this.newPosition = value;
  }

  @Override
  public Angle getMotorPosition() {
    return motorSim.getAngularPosition();
  }
}

// Copyright (c) FIRST and other WPILib contributors.
// Open Source Software; you can modify and/or share it under the terms of
// the WPILib BSD license file in the root directory of this project.

package frc.robot.systems.shooter;

import static edu.wpi.first.units.Units.Degrees;
import static edu.wpi.first.units.Units.Inches;
import static edu.wpi.first.units.Units.Meters;

import com.chargerrobotics.common.subsystem.belted_intake.BeltedIntakeSubsystem;
import com.chargerrobotics.common.subsystem.belted_intake.BeltedIntakeSubsystem.BeltedIntakeSubsystemState;
import edu.wpi.first.math.VecBuilder;
import edu.wpi.first.math.Vector;
import edu.wpi.first.math.geometry.Pose2d;
import edu.wpi.first.math.geometry.Rotation2d;

import edu.wpi.first.math.interpolation.InterpolatingTreeMap;
import edu.wpi.first.math.kinematics.ChassisSpeeds;
import edu.wpi.first.math.numbers.N3;
import edu.wpi.first.units.Units;
import edu.wpi.first.units.measure.Angle;

import edu.wpi.first.wpilibj2.command.SubsystemBase;
import frc.robot.systems.shooter.flywheel.FlywheelSubsystem;
import frc.robot.systems.shooter.flywheel.FlywheelSubsystem.FlywheelSubsystemState;
import frc.robot.systems.shooter.hood.HoodSubsystem;
import frc.robot.systems.shooter.hood.HoodSubsystem.HoodSubsystemStates;
import frc.robot.systems.shooter.turret.TurretSubsystem;
import frc.robot.systems.shooter.turret.TurretSubsystem.TurretState;
import frc.robot.utils.FeatureFlags;
import frc.robot.utils.RobotFeatureFlags;
import java.util.function.Supplier;
import lombok.Builder;
import lombok.Getter;
import lombok.NonNull;
import lombok.Setter;
import org.littletonrobotics.junction.Logger;

/**
 * The Shooter System for the 2026 Robot. This class manages the logic and handling of all Shooter
 * related subsystems
 */
public class ShooterSystem extends SubsystemBase {
  /** The states that the {@link ShooterSystem} can reside in */
  public enum ShooterState {
    IDLE,
    AIM,
    SHOOT
  }

  public enum ShooterStateInternal {
    /** Represents the state for the Shooter System to be attempting to achieve nothing */
    IDLE,
    /**
     * Represents the state for the Shooter System to be actively preparing to be targeting at a
     * point
     */
    AIMING,
    /** Represents the state for the Shooter System to be aimed at a target, and ready to shoot */
    AIMED,
    /** Represents the state for the Shooter System to be actively firing fuel */
    SHOOTING
  }

  // Interpolation Map Row Keys
  private static final int FLYWHEEL_VELOCITY_INTERPOLATED_MAP_ROW = 0;
  private static final int HOOD_ANGLE_INTERPOLATED_MAP_ROW = 1;
  private static final int TIME_OF_FLIGHT_INTERPOLATED_MAP_ROW = 2;

  private static final int TIME_OF_FLIGHT_CALCULATION_ITERATIONS = 3;

  // From the robot's center to the shooter's center
  private static final Translation2d SHOOTER_OFFSET =
      new Translation2d(Inches.of(0), Inches.of(5.56));
  private static final double RADIUS = SHOOTER_OFFSET.getNorm();
  private static final Rotation2d SHOOTER_ANGLE = SHOOTER_OFFSET.getAngle();

  @Getter private ShooterStateInternal currentState;
  @Setter private ShooterState desiredState;
  @Getter private Angle turretAngleToTarget;
  @Getter private Angle hoodAngleToTarget;
  @Getter private AngularVelocity flywheelVelocityToTarget;
  private Supplier<Pose2d> robotPositionSupplier;
  private Supplier<Translation2d> targetGoalPositionSupplier;
  private Supplier<ChassisSpeeds> robotSpeedSupplier;
  private TurretSubsystem turretSubsystem;
  private FlywheelSubsystem flywheelSubsystem;
  private BeltedIntakeSubsystem kickerSubsystem;
  private HoodSubsystem hoodSubsystem;

  /**
   * The math for the interpolation https://www.desmos.com/calculator/rwshoclimw
   *
   * @apiNote Make sure flywheel speed is stored in n1, while hood angle stored in n2
   * @see {@link ShooterSystem#FLYWHEEL_VELOCITY_INTERPOLATED_MAP_ROW} and {@link
   *     ShooterSystem#HOOD_ANGLE_INTERPOLATED_MAP_ROW} for the rows of the map for the
   *     corresponding value
   */
  @Setter private InterpolatingTreeMap<Distance, Vector<N3>> shotInterpolationTree;

  /** Creates a new ShooterSystem. */
  @Builder
  private ShooterSystem(
      @NonNull Supplier<Pose2d> robotPosition,
      @NonNull Supplier<Translation2d> targetGoalPosition,
      @NonNull Supplier<ChassisSpeeds> robotSpeed,
      @NonNull TurretSubsystem turretSubsystem,
      @NonNull FlywheelSubsystem flywheelSubsystem,
      @NonNull HoodSubsystem hoodSubsystem,
      @NonNull BeltedIntakeSubsystem kickerSubsystem) {

    this.robotPositionSupplier = robotPosition;
    this.targetGoalPositionSupplier = targetGoalPosition;
    this.robotSpeedSupplier = robotSpeed;
    this.turretSubsystem = turretSubsystem;
    this.flywheelSubsystem = flywheelSubsystem;
    this.hoodSubsystem = hoodSubsystem;
    this.kickerSubsystem = kickerSubsystem;

    this.desiredState = ShooterState.IDLE;
    this.currentState = ShooterStateInternal.IDLE;
    this.turretAngleToTarget = Units.Degrees.of(0);
    this.shotInterpolationTree =
        new InterpolatingTreeMap<>(
            (start, end, q) -> q.minus(start).div(end.minus(start)).magnitude(),
            (start, end, t) -> start.plus(end.minus(start).times(t)));

    this.shotInterpolationTree.put(
        Units.Meters.of(0),
        VecBuilder.fill(
            Units.RPM.of(0).magnitude(),
            Units.Degrees.of(0).magnitude(),
            Units.Seconds.of(0).magnitude())); // Replace me with actual data

    this.shotInterpolationTree.put(
        Units.Meters.of(5),
        VecBuilder.fill(
            Units.RPM.of(6000).magnitude(),
            Units.Degrees.of(90).magnitude(),
            Units.Seconds.of(5).magnitude())); // Replace me with actual data
  }

  // Runs 50 times every second (roughly) and drives the state machine.
  @Override
  public void periodic() {
    findCurrentState();
    switch (desiredState) {
      case IDLE -> handleIdleState();
      case AIM -> handleAimState();
      case SHOOT -> handleShootState();
    }

    Logger.recordOutput("ShooterSystem/CurrentState", currentState);
    Logger.recordOutput("ShooterSystem/DesiredState", desiredState);
  }

  /** Handles the Shooter System if the desired state is {@link ShooterState#IDLE} */
  private void handleIdleState() {
    flywheelSubsystem.setMotorSpeed(Units.RPM.of(0));
    kickerSubsystem.setDesiredState(BeltedIntakeSubsystemState.IDLE);
  }

  /** Handles the Shooter System if the desired state is {@link ShooterState#AIM} */
  private void handleAimState() {
    /*
     * Calculate and set turret target setpoint
     * Calculate and set flywheel speed
     * Calculate and set hood angle
     */
    Translation2d targetGoalPosition = targetGoalPositionSupplier.get();
    // The goal taking account for robot velocity
    Translation2d virtualGoalPosition = targetGoalPosition;
    Pose2d robotPosition = robotPositionSupplier.get();
    Distance distanceFromTarget =
        Units.Meters.of(targetGoalPosition.getDistance(robotPosition.getTranslation()));

    Vector<N3> shotCalculation = shotInterpolationTree.get(distanceFromTarget);

    if (RobotFeatureFlags.DATA_STORE.isEnabled(FeatureFlags.FUTURE_SHOT_CALCULATION)) {
      ChassisSpeeds robotSpeed = robotSpeedSupplier.get();
      Translation2d rotationalVelocity =
          new Translation2d(0, (RADIUS * robotSpeed.omegaRadiansPerSecond))
              .rotateBy(robotPosition.getRotation().plus(SHOOTER_ANGLE));
      Translation2d shooterVelocity =
          rotationalVelocity.plus(
              new Translation2d(robotSpeed.vxMetersPerSecond, robotSpeed.vyMetersPerSecond));

      for (int i = 0; i < TIME_OF_FLIGHT_CALCULATION_ITERATIONS; i++) {
        shotCalculation =
            shotInterpolationTree.get(
                Meters.of(virtualGoalPosition.getDistance(robotPosition.getTranslation())));
        virtualGoalPosition =
            targetGoalPosition.minus(
                shooterVelocity.times(shotCalculation.get(TIME_OF_FLIGHT_INTERPOLATED_MAP_ROW)));
      }
    }

    flywheelVelocityToTarget =
        Units.RPM.of(shotCalculation.get(FLYWHEEL_VELOCITY_INTERPOLATED_MAP_ROW));
    hoodAngleToTarget = Units.Degrees.of(shotCalculation.get(HOOD_ANGLE_INTERPOLATED_MAP_ROW));
    flywheelSubsystem.setMotorSpeed(flywheelVelocityToTarget);
    hoodSubsystem.setDesiredAngle(hoodAngleToTarget);

    turretAngleToTarget =
        Degrees.of(
            virtualGoalPosition
                .minus(robotPosition.getTranslation())
                .getAngle()
                .minus(robotPosition.getRotation())
                .getMeasure()
                .in(Degrees));

    turretSubsystem.setDesiredAngle(turretAngleToTarget);
    Logger.recordOutput("ShooterSystem/ShooterTarget", targetGoalPosition);
    Logger.recordOutput("ShooterSystem/CalculatedShooterTarget", virtualGoalPosition);
    Logger.recordOutput("ShooterSystem/FlywheelSetpointCalculation", flywheelVelocityToTarget);
    Logger.recordOutput("ShooterSystem/HoodSetpointCalculation", hoodAngleToTarget);
    Logger.recordOutput("ShooterSystem/TurretSetpointCalculation", turretAngleToTarget);
  }

  private void handleShootState() {
    // Make sure we are actually aimed before attempting to fire.
    handleAimState();
    if (flywheelSubsystem.getCurrentState() == FlywheelSubsystemState.READY
        && turretSubsystem.getCurrentState() == TurretState.AT_SETPOINT
        && hoodSubsystem.getCurrentState() == HoodSubsystemStates.AT_SETPOINT) {
      kickerSubsystem.setDesiredState(BeltedIntakeSubsystemState.INTAKE);
    } else {
      kickerSubsystem.setDesiredState(BeltedIntakeSubsystemState.STOPPED);
    }
  }

  /** Finds the current state that the Shooter System is currently in */
  private void findCurrentState() {
    TurretState currentTurretState = turretSubsystem.getCurrentState();
    FlywheelSubsystem.FlywheelSubsystemState currentFlywheelState =
        flywheelSubsystem.getCurrentState();
    HoodSubsystem.HoodSubsystemStates currentHoodState = hoodSubsystem.getCurrentState();
    BeltedIntakeSubsystemState currentKickerState = kickerSubsystem.getCurrentState();
    /*
     * Will set a IDLE current state:
     *  Turret is IDLE
     *  Flywheel is IDLE
     *  Kicker is STOPPED
     *  Hood is IDLE
     */
    if (currentTurretState == TurretState.IDLE
        && currentFlywheelState == FlywheelSubsystemState.IDLE
        && currentKickerState == BeltedIntakeSubsystemState.IDLE
        && currentHoodState == HoodSubsystemStates.IDLE) {
      currentState = ShooterStateInternal.IDLE;
    }

    /*
    * Will set a AIMING current state
    *   Turret is MOVING_TO_SETPOINT
        or
        Hood is MOVING_TO_SETPOINT
    */
    else if (currentTurretState == TurretState.MOVING_TO_SETPOINT
        || currentHoodState == HoodSubsystemStates.MOVING_TO_SETPOINT
        || currentFlywheelState == FlywheelSubsystem.FlywheelSubsystemState.ACTIVATING) {
      currentState = ShooterStateInternal.AIMING;
    }

    /**
     * Will set a AIMED current state Turret is AT_SETPOINT Hood is AT_SETPOINT Flywheel is READY
     * Kicker is STOPPED
     */
    else if (currentTurretState == TurretState.AT_SETPOINT
        && currentHoodState == HoodSubsystemStates.AT_SETPOINT
        && currentFlywheelState == FlywheelSubsystemState.READY
        && currentKickerState == BeltedIntakeSubsystemState.STOPPED) {
      currentState = ShooterStateInternal.AIMED;
    }

    /*
    * Will set a SHOOTING current state
        Turret is AT_SETPOINT
        Hood is AT_SETPOINT
        Flywheel is READY
        Kicker is INTAKE
    */
    else if (currentTurretState == TurretState.AT_SETPOINT
        && currentFlywheelState == FlywheelSubsystemState.READY
        && currentKickerState == BeltedIntakeSubsystemState.INTAKE
        && currentHoodState == HoodSubsystemStates.AT_SETPOINT) {
      currentState = ShooterStateInternal.SHOOTING;
    }
  }
}

import com.revrobotics.spark.config.ClosedLoopConfig;
import com.revrobotics.spark.config.FeedForwardConfig;
import edu.wpi.first.networktables.DoubleSubscriber;
import edu.wpi.first.networktables.DoubleTopic;
import edu.wpi.first.networktables.NetworkTable;
import edu.wpi.first.networktables.NetworkTableEvent.Kind;
import edu.wpi.first.networktables.NetworkTableInstance;
import java.util.EnumSet;
import java.util.function.Consumer;

public class NetworkTableClosedLoopTuner {
  private NetworkTableClosedLoopTuner() {}

  public static void createTuneableSparkPidController(
      String motorName, Consumer<ClosedLoopConfig> motorApplyConfig) {
    NetworkTableInstance networkTableInstance = NetworkTableInstance.getDefault();
    NetworkTable networkTable = networkTableInstance.getTable("PidTuner").getSubTable(motorName);

    DoubleTopic pTopic = networkTable.getDoubleTopic("P");
    DoubleTopic iTopic = networkTable.getDoubleTopic("I");
    DoubleTopic dTopic = networkTable.getDoubleTopic("D");

    pTopic.publish();
    iTopic.publish();
    dTopic.publish();

    DoubleSubscriber dSub = dTopic.subscribe(0);
    DoubleSubscriber iSub = iTopic.subscribe(0);
    DoubleSubscriber pSub = pTopic.subscribe(0);

    NetworkTableInstance.getDefault()
        .addListener(
            pSub,
            EnumSet.of(Kind.kValueAll),
            event ->
                motorApplyConfig.accept(
                    new ClosedLoopConfig()
                        .pid(event.valueData.value.getDouble(), iSub.get(), dSub.get())));

    NetworkTableInstance.getDefault()
        .addListener(
            iSub,
            EnumSet.of(Kind.kValueAll),
            event ->
                motorApplyConfig.accept(
                    new ClosedLoopConfig()
                        .pid(pSub.get(), event.valueData.value.getDouble(), dSub.get())));

    NetworkTableInstance.getDefault()
        .addListener(
            dSub,
            EnumSet.of(Kind.kValueAll),
            event ->
                motorApplyConfig.accept(
                    new ClosedLoopConfig()
                        .pid(pSub.get(), iSub.get(), event.valueData.value.getDouble())));
  }

  public static void createTuneableSparkFeedForwardController(
      String motorName, Consumer<FeedForwardConfig> motorApplyConfig) {
    NetworkTable networkTable =
        NetworkTableInstance.getDefault().getTable("FeedForwardTuner").getSubTable(motorName);

    DoubleTopic sTopic = networkTable.getDoubleTopic("S");
    DoubleTopic vTopic = networkTable.getDoubleTopic("V");
    DoubleTopic aTopic = networkTable.getDoubleTopic("A");
    DoubleTopic gTopic = networkTable.getDoubleTopic("G");

    sTopic.publish();
    vTopic.publish();
    aTopic.publish();
    gTopic.publish();

    DoubleSubscriber sSub = sTopic.subscribe(0);
    DoubleSubscriber vSub = vTopic.subscribe(0);
    DoubleSubscriber aSub = aTopic.subscribe(0);
    DoubleSubscriber gSub = gTopic.subscribe(0);

    NetworkTableInstance.getDefault()
        .addListener(
            sSub,
            EnumSet.of(Kind.kValueAll),
            event ->
                motorApplyConfig.accept(
                    new FeedForwardConfig()
                        .svag(
                            event.valueData.value.getDouble(),
                            vSub.get(),
                            aSub.get(),
                            gSub.get())));
    NetworkTableInstance.getDefault()
        .addListener(
            vSub,
            EnumSet.of(Kind.kValueAll),
            event ->
                motorApplyConfig.accept(
                    new FeedForwardConfig()
                        .svag(
                            sSub.get(),
                            event.valueData.value.getDouble(),
                            aSub.get(),
                            gSub.get())));
    NetworkTableInstance.getDefault()
        .addListener(
            aSub,
            EnumSet.of(Kind.kValueAll),
            event ->
                motorApplyConfig.accept(
                    new FeedForwardConfig()
                        .svag(
                            sSub.get(),
                            vSub.get(),
                            event.valueData.value.getDouble(),
                            gSub.get())));
    NetworkTableInstance.getDefault()
        .addListener(
            gSub,
            EnumSet.of(Kind.kValueAll),
            event ->
                motorApplyConfig.accept(
                    new FeedForwardConfig()
                        .svag(
                            sSub.get(),
                            vSub.get(),
                            aSub.get(),
                            event.valueData.value.getDouble())));
  }
}

      setUseTiming(false);

      CommandScheduler.getInstance().schedule(autonomousCommand);
    }
  }

  @Override
  public void teleopInit() {
    if (autonomousCommand != null) {
      autonomousCommand.cancel();
    }
  }

  @Override
  public void testInit() {
    CommandScheduler.getInstance().cancelAll();
  }
}

  Darwin* )         darwin=true  ;; #(
  MSYS* | MINGW* )  msys=true    ;; #(
  NONSTOP* )        nonstop=true ;;
esac



# Determine the Java command to use to start the JVM.
if [ -n "$JAVA_HOME" ] ; then
    if [ -x "$JAVA_HOME/jre/sh/java" ] ; then
        # IBM's JDK on AIX uses strange locations for the executables
        JAVACMD=$JAVA_HOME/jre/sh/java
    else
        JAVACMD=$JAVA_HOME/bin/java
    fi
    if [ ! -x "$JAVACMD" ] ; then
        die "ERROR: JAVA_HOME is set to an invalid directory: $JAVA_HOME

Please set the JAVA_HOME variable in your environment to match the
location of your Java installation."
    fi
else
    JAVACMD=java
    if ! command -v java >/dev/null 2>&1
    then
        die "ERROR: JAVA_HOME is not set and no 'java' command could be found in your PATH.

Please set the JAVA_HOME variable in your environment to match the
location of your Java installation."
    fi
fi

# Increase the maximum file descriptors if we can.
if ! "$cygwin" && ! "$darwin" && ! "$nonstop" ; then
    case $MAX_FD in #(
      max*)
        # In POSIX sh, ulimit -H is undefined. That's why the result is checked to see if it worked.
        # shellcheck disable=SC2039,SC3045
        MAX_FD=$( ulimit -H -n ) ||
            warn "Could not query maximum file descriptor limit"
    esac
    case $MAX_FD in  #(
      '' | soft) :;; #(
      *)
        # In POSIX sh, ulimit -n is undefined. That's why the result is checked to see if it worked.
        # shellcheck disable=SC2039,SC3045
        ulimit -n "$MAX_FD" ||
            warn "Could not set maximum file descriptor limit to $MAX_FD"
    esac
fi

# Collect all arguments for the java command, stacking in reverse order:
#   * args from the command line
#   * the main class name
#   * -classpath
#   * -D...appname settings
#   * --module-path (only if needed)
#   * DEFAULT_JVM_OPTS, JAVA_OPTS, and GRADLE_OPTS environment variables.

# For Cygwin or MSYS, switch paths to Windows format before running java
if "$cygwin" || "$msys" ; then
    APP_HOME=$( cygpath --path --mixed "$APP_HOME" )

    JAVACMD=$( cygpath --unix "$JAVACMD" )

    # Now convert the arguments - kludge to limit ourselves to /bin/sh
    for arg do
        if
            case $arg in                                #(
              -*)   false ;;                            # don't mess with options #(
              /?*)  t=${arg#/} t=/${t%%/*}              # looks like a POSIX filepath
                    [ -e "$t" ] ;;                      #(
              *)    false ;;
            esac
        then
            arg=$( cygpath --path --ignore --mixed "$arg" )
        fi
        # Roll the args list around exactly as many times as the number of
        # args, so each arg winds up back in the position where it started, but
        # possibly modified.
        #
        # NB: a `for` loop captures its iteration list before it begins, so
        # changing the positional parameters here affects neither the number of
        # iterations, nor the values presented in `arg`.
        shift                   # remove old arg
        set -- "$@" "$arg"      # push replacement arg
    done
fi


# Add default JVM options here. You can also use JAVA_OPTS and GRADLE_OPTS to pass JVM options to this script.
DEFAULT_JVM_OPTS='"-Xmx64m" "-Xms64m"'

# Collect all arguments for the java command:
#   * DEFAULT_JVM_OPTS, JAVA_OPTS, and optsEnvironmentVar are not allowed to contain shell fragments,
#     and any embedded shellness will be escaped.
#   * For example: A user cannot expect ${Hostname} to be expanded, as it is an environment variable and will be
#     treated as '${Hostname}' itself on the command line.

set -- \
        "-Dorg.gradle.appname=$APP_BASE_NAME" \
        -jar "$APP_HOME/gradle/wrapper/gradle-wrapper.jar" \
        "$@"

# Stop when "xargs" is not available.
if ! command -v xargs >/dev/null 2>&1
then
    die "xargs is not available"
fi

# Use "xargs" to parse quoted args.
#
# With -n1 it outputs one arg per line, with the quotes and backslashes removed.
#
# In Bash we could simply go:
#
#   readarray ARGS < <( xargs -n1 <<<"$var" ) &&
#   set -- "${ARGS[@]}" "$@"
#
# but POSIX shell has neither arrays nor command substitution, so instead we
# post-process each arg (as a line of input to sed) to backslash-escape any
# character that might be a shell metacharacter, then use eval to reverse
# that process (while maintaining the separation between arguments), and wrap
# the whole thing up as a single "set" statement.
#
# This will of course break if any of these variables contains a newline or
# an unmatched quote.
#

eval "set -- $(
        printf '%s\n' "$DEFAULT_JVM_OPTS $JAVA_OPTS $GRADLE_OPTS" |
        xargs -n1 |
        sed ' s~[^-[:alnum:]+,./:=@_]~\\&~g; ' |
        tr '\n' ' '
    )" '"$@"'

exec "$JAVACMD" "$@"

```
